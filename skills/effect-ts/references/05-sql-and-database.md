# 5. SQL & Database

Effect provides `@effect/sql` for type-safe database access with models, repositories, migrations, and transactions.

## Setup

### Install Dependencies

```bash
# For SQLite
pnpm add @effect/sql @effect/sql-sqlite-node

# For PostgreSQL
pnpm add @effect/sql @effect/sql-pg
```

### Configure the Database Layer

```typescript
// Sql.ts
import { NodeContext } from "@effect/platform-node"
import { SqlClient } from "@effect/sql"
import { SqliteClient, SqliteMigrator } from "@effect/sql-sqlite-node"
import { identity, Layer } from "effect"
import { fileURLToPath } from "url"

// Database connection
const ClientLive = SqliteClient.layer({
  filename: "data/db.sqlite"
})

// Auto-run migrations on startup
const MigratorLive = SqliteMigrator.layer({
  loader: SqliteMigrator.fromFileSystem(
    fileURLToPath(new URL("./migrations", import.meta.url))
  )
}).pipe(Layer.provide(NodeContext.layer))

// Combined layer: migrates then provides the client
export const SqlLive = MigratorLive.pipe(
  Layer.provideMerge(ClientLive)
)
```

## Domain Models with `Model.Class`

`Model.Class` extends `Schema.Class` with database-specific field types:

```typescript
import { Model } from "@effect/sql"
import { Schema } from "effect"

// Branded ID type
export const AccountId = Schema.Number.pipe(Schema.brand("AccountId"))
export type AccountId = typeof AccountId.Type

export class Account extends Model.Class<Account>("Account")({
  id: Model.Generated(AccountId),       // Auto-generated by DB
  createdAt: Model.DateTimeInsert,       // Auto-set on insert
  updatedAt: Model.DateTimeUpdate        // Auto-set on update
}) {}
```

### Model Field Types

| Field Type | Behavior |
|------------|----------|
| `Model.Generated(schema)` | Auto-generated by DB (excluded from insert type) |
| `Model.GeneratedByApp(schema)` | Generated by application code (included in insert type) |
| `Model.DateTimeInsert` | DateTime, auto-set on insert |
| `Model.DateTimeUpdate` | DateTime, auto-updated |
| `Model.Sensitive(schema)` | Excluded from JSON serialization by default |
| `Model.FieldOption(schema)` | Optional field (maps to `Option`) |
| `Model.Date` | Date-only field |

### Full Example: User Model

```typescript
import { Model } from "@effect/sql"
import { Schema } from "effect"
import { AccessToken } from "./AccessToken.js"
import { AccountId } from "./Account.js"
import { Email } from "./Email.js"

export const UserId = Schema.Number.pipe(Schema.brand("UserId"))
export type UserId = typeof UserId.Type

export class User extends Model.Class<User>("User")({
  id: Model.Generated(UserId),
  accountId: Model.GeneratedByApp(AccountId),
  email: Email,
  accessToken: Model.Sensitive(AccessToken),  // Hidden from JSON
  createdAt: Model.DateTimeInsert,
  updatedAt: Model.DateTimeUpdate
}) {}
```

### Model with Sensitive Data Exposed

```typescript
export class UserWithSensitive extends Model.Class<UserWithSensitive>(
  "UserWithSensitive"
)({
  ...Model.fields(User),           // Spread all fields from User
  accessToken: AccessToken,        // Override: show the token
  account: Account                 // Add related data
}) {}
```

### Model with Optional Fields

```typescript
export class Person extends Model.Class<Person>("Person")({
  id: Model.Generated(PersonId),
  groupId: Model.GeneratedByApp(GroupId),
  firstName: Schema.NonEmptyTrimmedString,
  lastName: Schema.NonEmptyTrimmedString,
  dateOfBirth: Model.FieldOption(Model.Date),  // Optional date
  createdAt: Model.DateTimeInsert,
  updatedAt: Model.DateTimeUpdate
}) {}
```

## Auto-Generated Types

Each `Model.Class` automatically generates several types:

```typescript
// The full type (for reading from DB)
type Account = Account

// Insert type (without Generated fields)
Account.insert        // Schema for inserts
Account.insert.make({ /* only non-generated fields */ })

// JSON types (for API)
Account.json          // Schema for serialization
Account.jsonCreate    // Schema for creation payloads
Account.jsonUpdate    // Schema for update payloads
```

## Repositories

### Basic Repository with `Model.makeRepository`

```typescript
import { Model } from "@effect/sql"
import { Effect } from "effect"
import { Group } from "../Domain/Group.js"
import { SqlLive } from "../Sql.js"

export class GroupsRepo extends Effect.Service<GroupsRepo>()("Groups/Repo", {
  effect: Model.makeRepository(Group, {
    tableName: "groups",
    spanPrefix: "GroupsRepo",
    idColumn: "id"
  }),
  dependencies: [SqlLive]
}) {}
```

`Model.makeRepository` automatically provides:
- `insert(record)` — Insert a new record
- `findById(id)` — Find by primary key (returns `Option`)
- `update(record)` — Update a record
- `delete(id)` — Delete by primary key

### Extended Repository (Custom Queries)

```typescript
import { Model, SqlClient, SqlSchema } from "@effect/sql"
import { Effect, pipe } from "effect"

export class UsersRepo extends Effect.Service<UsersRepo>()(
  "Accounts/UsersRepo",
  {
    effect: Effect.gen(function*() {
      const sql = yield* SqlClient.SqlClient

      // Standard CRUD from makeRepository
      const repo = yield* Model.makeRepository(User, {
        tableName: "users",
        spanPrefix: "UsersRepo",
        idColumn: "id"
      })

      // Custom query using SqlSchema
      const findByAccessTokenSchema = SqlSchema.findOne({
        Request: AccessToken,
        Result: User,
        execute: (key) =>
          sql`select * from users where accessToken = ${key}`
      })

      const findByAccessToken = (apiKey: AccessToken) =>
        pipe(
          findByAccessTokenSchema(apiKey),
          Effect.orDie,
          Effect.withSpan("UsersRepo.findByAccessToken")
        )

      return { ...repo, findByAccessToken } as const
    }),
    dependencies: [SqlLive]
  }
) {}
```

## Raw SQL Queries

### Using the `SqlClient`

```typescript
const sql = yield* SqlClient.SqlClient

// Template literal queries (auto-parameterized)
yield* sql`SELECT * FROM users WHERE email = ${email}`

// Dialect-specific SQL
yield* sql.onDialectOrElse({
  pg: () => sql`CREATE TABLE accounts (
    id SERIAL PRIMARY KEY,
    createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL
  )`,
  orElse: () => sql`CREATE TABLE accounts (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    createdAt DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL
  )`
})
```

### `SqlSchema` for Type-Safe Queries

```typescript
import { SqlSchema } from "@effect/sql"

// findOne — returns Option<Result>
const findByEmail = SqlSchema.findOne({
  Request: Email,
  Result: User,
  execute: (email) => sql`SELECT * FROM users WHERE email = ${email}`
})

// findAll — returns Array<Result>
const findAllUsers = SqlSchema.findAll({
  Request: Schema.Void,
  Result: User,
  execute: () => sql`SELECT * FROM users`
})
```

## Transactions

```typescript
const sql = yield* SqlClient.SqlClient

const createUserWithAccount = (userData: UserInput) =>
  accountRepo.insert(Account.insert.make({})).pipe(
    Effect.bind("user", (account) =>
      userRepo.insert(User.insert.make({
        ...userData,
        accountId: account.id
      }))
    ),
    sql.withTransaction,  // ← Wraps everything in a DB transaction
    Effect.orDie
  )
```

Transactions are also used in "with" patterns:

```typescript
const with_ = <A, E, R>(
  id: GroupId,
  f: (group: Group) => Effect.Effect<A, E, R>
): Effect.Effect<A, E | GroupNotFound, R> =>
  pipe(
    repo.findById(id),
    Effect.flatMap(
      Option.match({
        onNone: () => new GroupNotFound({ id }),
        onSome: Effect.succeed
      })
    ),
    Effect.flatMap(f),
    sql.withTransaction,                                    // Transaction
    Effect.catchTag("SqlError", (err) => Effect.die(err)),  // Convert SQL errors to defects
    Effect.withSpan("Groups.with", { attributes: { id } })
  )
```

## Migrations

### File-Based Migrations

Place migration files in `src/migrations/` numbered sequentially:

```
src/migrations/
  00001_create accounts.ts
  00002_create groups.ts
  00003_create people.ts
```

### Migration File Structure

```typescript
// src/migrations/00001_create accounts.ts
import { SqlClient } from "@effect/sql"
import { Effect } from "effect"

export default Effect.gen(function*() {
  const sql = yield* SqlClient.SqlClient

  yield* sql.onDialectOrElse({
    pg: () => sql`
      CREATE TABLE accounts (
        id SERIAL PRIMARY KEY,
        createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
        updatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL
      )
    `,
    orElse: () => sql`
      CREATE TABLE accounts (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        createdAt DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL,
        updatedAt DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL
      )
    `
  })
})
```

### Migration with Foreign Keys

```typescript
// src/migrations/00002_create groups.ts
export default Effect.gen(function*() {
  const sql = yield* SqlClient.SqlClient

  yield* sql.onDialectOrElse({
    pg: () => sql`
      CREATE TABLE groups (
        id SERIAL PRIMARY KEY,
        ownerId INTEGER NOT NULL,
        name VARCHAR(255) NOT NULL,
        createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
        updatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
        FOREIGN KEY (ownerId) REFERENCES accounts(id)
      )
    `,
    orElse: () => sql`
      CREATE TABLE groups (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        ownerId INTEGER NOT NULL,
        name TEXT NOT NULL,
        createdAt DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL,
        updatedAt DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL,
        FOREIGN KEY (ownerId) REFERENCES accounts(id)
      )
    `
  })
})
```

## Test Mocking for SQL

Instead of using a real database in tests, mock the `SqlClient`:

```typescript
import { makeTestLayer } from "./lib/Layer.js"

export const SqlTest = makeTestLayer(SqlClient.SqlClient)({
  withTransaction: identity  // No-op transaction in tests
})

// Use in service test layers:
export class Accounts extends Effect.Service<Accounts>()("Accounts", {
  /* ... */
}) {
  static Test = this.DefaultWithoutDependencies.pipe(
    Layer.provideMerge(SqlTest),
    Layer.provideMerge(Uuid.Test)
  )
}
```
